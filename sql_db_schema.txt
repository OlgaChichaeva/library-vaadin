
CREATE CACHED table author(
id_author BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
surname varchar(255) NOT NULL,
f_name varchar(255) NOT NULL,
l_name varchar(255)NOT NULL
);

CREATE CACHED table genre(
id_genre BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
genre varchar(255) NOT NULL
)

CREATE CACHED table book(
id_book BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
id_author BIGINT,
title varchar(255) NOT NULL,
id_genre BIGINT,
publisher varchar(255) NOT NULL,
year_of_issue varchar(255) NOT NULL,
city  varchar(255) NOT NULL,
CONSTRAINT fk_author_book foreign key(id_author) REFERENCES author(id_author),
CONSTRAINT fk_genre_book foreign key(id_genre) REFERENCES genre(id_genre)
);

-----------------------
CREATE index book_i$_id_autor on book(id_autor);
